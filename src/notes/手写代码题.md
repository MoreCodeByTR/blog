## 1.防抖
事件在触发n秒后执行回调，如果n秒内事件再次被触发，则重新进行计时
```js
function debounce(fn,delay){
    let timer=null

    return function(){
        let ctx=this
        let args=arguments

        if(timer){
            clearTimeout(timer)
            timer=null
        }

        timer=setTimeout(()=>{
            fn.apply(ctx,args)
        },delay)
    }
}

//测试用例
function show(info){
    console.log(info)
}

let deShow=debounce(show,2000)

deShow(123)
deShow(456)
deShow(789)  //789

```

## 2.节流
单位时间内事件被多次触发，但只执行一次回调，用以减少事件调用的次数，可以放在onscrool、onmousemove
```js
function throttle(fn,interval){
    let preTime=Date.now()

    return function(){
        let ctx=this
        let args=arguments

        if(Date.now()-preTime>interval){
            fn.apply(ctx,args)
            preTime=Date.now()
        }
    }

}

//测试用例

function show(){
    console.log(new Date().toLocaleString())
}

let thShow=throttle(show,2000)

setInterval(()=>{thShow()},500)
```

```js
function myInstanceof(left,right){
    let proto=Object.getPrototypeOf(left)
    let rightPrototype=right.prototype

    while(true){
        if(proto==null) return false
        if(proto==rightPrototype) return true

        proto=Object.getPrototypeOf(proto)
    }
}
```

## 7. 实现new操作符
在调用 new 的过程中会发生以上四件事情：
（1）首先创建了一个新的空对象
（2）设置原型，将对象的原型设置为函数的 prototype 对象。
（3）让函数的 this 指向这个对象，执行构造函数的代码（为这个新对象添加属性）
（4）判断函数的返回值类型，如果是值类型，返回创建的对象。如果是引用类型，就返回这个引用类型的对象。
```js
function myNew(){
    let newObj=null
    let constructor=Array.prototype.shift.call(arguments)
    let result=null

    if(typeof constructor !='function'){
        console.error('type erroe')
        return 
    }

    newObj=Object.create(constructor.prototype)

    result=constructor.apply(newObj,arguments)

    let flag=result&&(typeof result==='object'||typeof result==="function")

    return flag?result:newObj

}

//测试用例

function Animal(age){
    this.age=age||20
}

console.log(myNew(Animal))
```

```js
function myCurry(fn){
    let length=fn.length

        let subArgs=args.concat([...arguments])
```

```js
function shallowCopy(obj){
    if(typeof obj!=='object'|| !obj) return 

    let res=Array.isArray(obj)?[]:{}

    for(let key in obj){
       if(obj.hasOwnProperty(key)){
           res[key]=obj[key]

function deepClone(obj){
    if(!obj||typeof obj!=='object') return false

    let res=Array.isArray(obj)?[]:{}

    for(let key in obj){
        if(obj.hasOwnProperty(key)){
            res[key]=typeof obj[key]==='object'?deepClone(obj[key]):obj[key]
        }
    }

    return res
}
```

## 10.使用promise封装ajax
```js
function request(method,url){
    return new Promise((resolve,reject)=>{
        let xhr=new XMLHttpRequest()
        xhr.open(method,url)
        xhr.onreadystatechange=function(data){
            if(xhr.readyState!=4) return
            if(xhr.status==200){
               resolve(this.response)
            }else{
                reject(new Error(this.statusText))
            }
        }

        xhr.onerror=function(){
            reject(new Error(this.statusText))
        }

        xhr.setRequestHeader('Accept','application/json')
        xhr.send(null)
    })
}
```

## 14.LRU缓存
缓存空间有限，get取值时有则返回，没有则-1
put新增时，有则更新，无则添加
```js
var LRUCache = function(capacity) {
    this.map=new Map()
    this.capacity=capacity
};

/** 
 * @param {number} key
 * @return {number}
 */
LRUCache.prototype.get = function(key) {
    if(this.map.has(key)){
        let val=this.map.get(key)
        this.map.delete(key)
        this.map.set(key,val)
        return this.map.get(key)
    }else{
        return -1
    }
};

/** 
 * @param {number} key 
 * @param {number} value
 * @return {void}
 */
LRUCache.prototype.put = function(key, value) {
    if(this.map.has(key)){
        this.map.delete(key)
    }
      this.map.set(key,value)
      if(this.map.size>this.capacity){
          this.map.delete(this.map.keys().next().value)
      }
};
```

## 17.快排
平均时间复杂度O(nlogn),最坏时间复杂度O(n2),当中间partition每次均分左右数组时会出现较差情况。

有的写法是将nums[0]作为partition时，如果原数组几乎排好序，则算法复杂度出现最差情况，因为每次分割数组时，左数组长度为0
```js
function quickSort(nums){
  if(nums.length<2) return nums

  let leftArr=[]
  let rightArr=[]
  let mid=Math.floor((nums.length-1)/2)

  for(let i=0;i<nums.length;i++){
    if(i==mid) continue
    if(nums[i]<=nums[mid]){
      leftArr.push(nums[i])
    }else{
      rightArr.push(nums[i])
    }
  }
  //每次递归能确定中间元素nums[mid]位置
  return quickSort(leftArr).concat(nums[mid],quickSort(rightArr))

   //   return   [...quickSort(leftArr),nums[mid],...quickSort(rightArr)]

}
```

```js
function mergeSort(nums){
  //边界条件
  if(nums.length<2) return nums

  let mid=Math.floor((nums.length)/2)

   //中间截取
  let leftArr=nums.slice(0,mid)
  let rightArr=nums.slice(mid)

  //由内到外合并
  return merge(mergeSort(leftArr),mergeSort(rightArr))


}

function merge(left,right){
  let res=[]
  while(left.length&&right.length){
    if(left[0]<=right[0]){
      res.push(left.shift())
    }else{
      res.push(right.shift())
    }
  }

  while(left.length) res.push(left.shift())
  while(right.length) res.push(right.shift())

  return res

}
```
